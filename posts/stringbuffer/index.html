  public static void bubbleSort(int[] arr) {
    for(int x=0; x<arr.length-1; x++) {
      for(int y=0; y<arr.length-1-x; y++) {
        if(arr[y]>arr[y+1]) {
          int temp = arr[y];
          arr[y] = arr[y+1];
          arr[y+1] = temp;
        }
      }
    }
  }

public static void selectSort(int[] a)
{
    int min=0;
    int temp=0;
    if((a==null)||(a.length==0))
        return;
    for(int i=0;i<a.length-1;i++)
    {
        min=i;//无序区的最小数据数组下标
        for(int  j=i+1;j<a.length;j++)
        {
            //在无序区中找到最小数据并保存其数组下标
            if(a[j]<a[min])
            {
                min=j;
            }
        }
        //将最小元素放到本次循环的前端
        temp=a[i];
        a[i]=a[min];
        a[min]=temp;
    }
}

public static int getIndex(int[] arr,int value) {
  int max = arr.length-1;
  int min = 0;
  int mid = (max+min)/2;

  while(arr[mid] != value) {
    if(arr[mid] > value) {
      max = mid - 1;
    }else if(arr[mid] < value) {
      min = mid + 1;
    }

    if(max < min) {
      return -1;
    }

    mid = (max+min)/2;
  }

  return mid;
}
